Polymorphism: mappend

(<>) = `mappend`

"abc" <>  "def" = "abcdef"
("abc","xyz") <> ("ABC","XYZ") = ("abcABC","xyzXYZ")
3 <> 4 = ERROR which law? + or *
-- Monoid (N,+)
type Sum = Sum {getSum :: a}
(<>+) = getSum (Sum x <> Sum y)
3 <>+ 4 = 7
-- Monoid (N,*)
type Product = Product {getProduct :: a}
(<>*) = getProduct (Product x <> Product y)
3 <>* 4 = 12
